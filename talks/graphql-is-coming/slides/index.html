class Task(DjangoObjectType):
    class Meta:
        model = TaskModel

class Query(ObjectType):
    tasks = List(Task)

Authorization

class Query(ObjectType):

    tasks = List(Task)

    def resolve_tasks(self, args, context, info):

        if context.user.is_authenticated():
            return TaskModel.objects.all()
        else:
            return TaskModel.objects.none()
